### switch 语句

```c
#include <stdio.h>
#include<stdlib.h>
#include<time.h>

int main()
{
	int human;
	int pc;
	int judge;
	int retry;//是否需要继续进行
	srand(time(NULL));//生成任意序列的随机数

	do {
		pc = rand() % 3;//电脑出
		printf("输入你的手势 (0)石头 (1)剪刀 (2)布："); scanf("%d", &human);
		printf("我出 ---");//
		switch (human)//显示玩家的手势
		{
		case 0: printf("石头\n"); break;
		case 1: printf("剪刀\n"); break;
		case 2: printf("布\n"); break;
		}

		printf("\n电脑出 ---");
		switch (pc)//显示电脑的手势
		{
		case 0: printf("石头\n"); break;
		case 1: printf("剪刀\n"); break;
		case 2: printf("布\n"); break;
		}
		printf("\n");

		//胜负评判
		judge = (human - pc + 3) % 3;
		switch (judge)
		{
		case 0:printf("平局\n"); break;
		case 1:printf("玩家失败\n"); break;
		case 2:printf("玩家胜利\n"); break;
		}
		printf("retry? (0)---no ;(1)---yes    "); scanf("%d", &retry);
		printf("\n");
	} while (retry == 1);
	
		system("pause");
		return 0;
}
```

### 程序改良

```c
#include <stdio.h>
#include<stdlib.h>
#include<time.h>

int main()
{

	int human;
	int pc;
	int judge;
	int i, retry;
	srand(time(NULL));

	char *hd[] = { "石头", "剪刀", "布" };

	do {
		pc = rand() % 3;//电脑输入
		do {
			for (i = 0; i < 3; i++)
				printf("(%d)%s  ", i, hd[i]);//(0)石头  (1)剪刀  (2)布
			printf(": "); scanf("%d", &human);
		} while (human < 0 || human > 2);//输入0-2的数字；否则重新输入

		printf("你出：%s   电脑出：%s\n", hd[human], hd[pc]);
		printf("\n");

		//胜负评判
		judge = (human - pc + 3) % 3;
		switch (judge)
		{
		case 0:printf("平局\n"); break;
		case 1:printf("玩家失败\n"); break;
		case 2:printf("玩家胜利\n"); break;
		}
		printf("retry? (0)---no ;(1)---yes    "); scanf("%d", &retry);
		printf("\n");

	} while (retry);
	system("pause");
	return 0;
}
```
